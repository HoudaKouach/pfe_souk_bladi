 date_inscription : DATETIME
 photo_profil : VARCHAR(255)
 logo_cooperative : VARCHAR(255)
)

CLIENT (
 id_client : INT (PK)

 téléphone : VARCHAR(20)
 adresse : TEXT

 date_inscription : DATETIME
)

CATEGORIE (
 id_categorie : INT (PK)

)

PRODUIT (
 id_produit : INT (PK)
